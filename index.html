/* * --------------------------------------- * BACKGROUND TRANSPARAN (WATERMARK EFFECT) * --------------------------------------- */ /* 1. Terapkan gambar latar belakang ke seluruh halaman */ body { /* Ganti 'background-pattern.jpg' dengan nama file gambar Anda */ background-image: url('nama_file_gambar_background_anda.jpg'); background-size: cover; /* Pastikan gambar menutupi seluruh area */ background-repeat: no-repeat; background-attachment: fixed; /* Opsi: agar gambar tetap saat di-scroll */ /* * PENTING: Untuk membuat lapisan transparan di atas gambar, * kita akan menggunakan properti 'background' pada elemen pseudo ::before. */ position: relative; /* Diperlukan untuk elemen pseudo */ } /* 2. Tambahkan lapisan transparan (overlay) di atas gambar latar belakang */ body::before { content: ""; /* Harus ada untuk elemen pseudo */ position: fixed; /* Tetap di tempatnya */ top: 0; left: 0; width: 100%; height: 100%; /* * Tambahkan warna overlay semi-transparan (misalnya, hitam pekat 20%). * Angka terakhir (0.2) mengontrol transparansi (0.0 = full transparan, 1.0 = tidak transparan). */ background-color: rgba(0, 0, 0, 0.2); z-index: -1; /* Posisikan di bawah konten utama (teks, menu, dll.) */ } /* 3. Penyesuaian agar konten utama memiliki latar belakang agar mudah dibaca */ /* Pastikan konten utama memiliki latar belakang semi-solid agar teks di atasnya jelas */ .section, .hero { /* Latar belakang putih semi-transparan untuk konten */ background-color: rgba(255, 255, 255, 0.9); padding: 40px; border-radius: 8px; margin-bottom: 20px; } /* Header dan Footer juga perlu latar belakang yang jelas */ header, footer { background-color: #333; /* Warna gelap */ color: white; /* Tambahkan transparansi jika diperlukan, tapi disarankan solid agar navigasi mudah dilihat */ }
